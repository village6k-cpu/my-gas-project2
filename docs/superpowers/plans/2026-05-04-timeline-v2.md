```javascript
/**
 * 스케줄상세 상태 변경
 * @param {number} rowIndex - 대표 행 번호
 * @param {string} newStatus - 새 상태 (대기/반출중/반납완료/취소)
 * @param {string} [rowIndices] - 같은 세트 전체 행 JSON
 */
function updateScheduleStatus(rowIndex, newStatus, rowIndices) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('스케줄상세');
  if (!sheet) return { success: false, message: '스케줄상세 시트 없음' };

  var validStatuses = ['대기', '반출중', '반납완료', '취소'];
  if (validStatuses.indexOf(newStatus) < 0) {
    return { success: false, message: '유효하지 않은 상태: ' + newStatus };
  }

  var rows = [];
  if (rowIndices) {
    try { rows = JSON.parse(rowIndices); } catch(e) { rows = [rowIndex]; }
  } else {
    rows = [rowIndex];
  }

  rows.forEach(function(r) {
    var ri = Number(r);
    if (ri >= 2) {
      sheet.getRange(ri, 10).setValue(newStatus);  // J열 = 10번째 = 상태
    }
  });

  return { success: true, message: rows.length + '행 상태 변경 완료', newStatus: newStatus };
}
```

```javascript
function calcDateRange(items) {
  var now = new Date();
  var min = new Date(now.getTime() - 3 * 86400000);  // 3일 전
  var max = new Date(now.getTime() + 21 * 86400000);  // 21일 후
  items.forEach(function(item) {
    var s = new Date(item.start);
    var e = new Date(item.end);
    if (s < min) min = new Date(s.getTime() - 86400000);
    if (e > max) max = new Date(e.getTime() + 86400000);
  });
  state.dateRange.start = stripTime(min);
  state.dateRange.end = stripTime(max);
}

function stripTime(d) {
  return new Date(d.getFullYear(), d.getMonth(), d.getDate());
}

function renderDateHeader() {
  var header = document.getElementById('dateHeader');
  var html = '';
  var d = new Date(state.dateRange.start);
  var today = stripTime(new Date());
  var days = ['일','월','화','수','목','금','토'];
  while (d <= state.dateRange.end) {
    var isToday = d.getTime() === today.getTime();
    var isWeekend = d.getDay() === 0 || d.getDay() === 6;
    var cls = (isToday ? ' today' : '') + (isWeekend ? ' weekend' : '');
    var month = d.getDate() === 1 || d.getTime() === state.dateRange.start.getTime()
      ? (d.getMonth()+1) + '월' : '';
    html += '<div class="date-col' + cls + '">'
      + '<span class="month">' + month + '</span>'
      + '<span class="day">' + d.getDate() + '</span>'
      + '<span class="weekday">' + days[d.getDay()] + '</span>'
      + '</div>';
    d = new Date(d.getTime() + 86400000);
  }
  header.innerHTML = html;
  // CSS 변수로 컬럼 너비 설정
  document.documentElement.style.setProperty('--col-w', state.colWidth + 'px');
}
```

```javascript
function groupItems(items, mode) {
  var grouped = {};  // groupKey → { name, items: [] }
  items.forEach(function(item) {
    var key;
    if (mode === 'customer') {
      key = item.custName || '미지정';
    } else if (mode === 'status') {
      key = item.status || '기타';
    } else {
      // 기본: 세트별 (group id 기준)
      key = item.group;
    }
    if (!grouped[key]) {
      var name = key;
      if (mode === 'set') {
        var g = state.groups.find(function(g) { return g.id === key; });
        name = g ? g.content : key;
      }
      grouped[key] = { name: name, items: [] };
    }
    grouped[key].items.push(item);
  });
  return grouped;
}
```

```javascript
function filterItems(items) {
  var query = state.searchQuery.toLowerCase();
  return items.filter(function(item) {
    // 상태 필터
    var st = ['대기','반출중','반납완료','취소'].indexOf(item.status) >= 0 ? item.status : '기타';
    if (state.hiddenStatuses[st]) return false;
    // 검색 필터
    if (query) {
      var custMatch = (item.custName || '').toLowerCase().indexOf(query) >= 0;
      var setMatch = (item.세트명 || '').toLowerCase().indexOf(query) >= 0;
      if (!custMatch && !setMatch) return false;
    }
    return true;
  });
}
```

```javascript
function renderAll() {
  renderDateHeader();

  var filtered = filterItems(state.items);
  var grouped = groupItems(filtered, state.groupMode);

  var labelHtml = '';
  var gridHtml = '';
  var totalDays = daysBetween(state.dateRange.start, state.dateRange.end) + 1;
  var today = stripTime(new Date());
  var statCounts = { '대기':0, '반출중':0, '반납완료':0, '취소':0 };
  var todayOut = 0, todayIn = 0;

  Object.keys(grouped).sort().forEach(function(key) {
    var group = grouped[key];
    var collapsed = !!state.collapsedGroups[key];

    // 라벨: 그룹 헤더
    labelHtml += '<div class="label-group">'
      + '<div class="label-group-header' + (collapsed ? ' collapsed' : '') + '" data-group="' + key + '">'
      + '<span class="chevron">▼</span>'
      + escHtml(group.name)
      + '<span class="count">' + group.items.length + '</span>'
      + '</div>';

    // 그리드: 그룹 헤더 행 (빈 셀)
    gridHtml += '<div class="grid-group">'
      + '<div class="grid-group-header">' + emptyCells(totalDays, today) + '</div>';

    if (!collapsed) {
      group.items.forEach(function(item) {
        // 통계
        var st = ['대기','반출중','반납완료','취소'].indexOf(item.status) >= 0 ? item.status : '기타';
        if (statCounts[st] !== undefined) statCounts[st]++;

        var itemStart = stripTime(new Date(item.start));
        var itemEnd = stripTime(new Date(item.end));
        if (itemStart.getTime() === today.getTime()) todayOut++;
        if (itemEnd.getTime() === today.getTime()) todayIn++;

        // 라벨: 아이템 행
        labelHtml += '<div class="label-row">' + escHtml(item.장비명 || item.custName) + '</div>';

        // 그리드: 아이템 행 + 바
        var barLeft = daysBetween(state.dateRange.start, itemStart) * state.colWidth;
        var barWidth = (daysBetween(itemStart, itemEnd) + 1) * state.colWidth - 4;
        var barText = escHtml((item.세트명원본 || item.세트명 || '') + ' / ' + (item.custName || '')
          + ' / ' + (item.반출시간 || '') + '-' + (item.반납시간 || '')
          + (item.회차 > 1 ? '(' + item.회차 + '회차)' : ''));
        var barClass = 'timeline-bar bar-' + (st === '기타' ? '기타' : item.status);

        // 오늘 반출/반납 표시
        if (itemStart.getTime() === today.getTime()) barClass += ' bar-today-out';
        if (itemEnd.getTime() === today.getTime()) barClass += ' bar-today-in';

        gridHtml += '<div class="grid-row" data-item-id="' + item.id + '">'
          + emptyCells(totalDays, today)
          + '<div class="' + barClass + '" style="left:' + barLeft + 'px;width:' + Math.max(barWidth, 30) + 'px"'
          + ' data-item-id="' + item.id + '"'
          + ' title="' + barText + '">'
          + '<span class="resize-handle left"></span>'
          + barText
          + '<span class="resize-handle right"></span>'
          + '</div></div>';
      });
    }

    labelHtml += '</div>';  // .label-group 닫기
    gridHtml += '</div>';   // .grid-group 닫기
  });

  // 오늘 라인
  var todayOffset = daysBetween(state.dateRange.start, today) * state.colWidth + state.colWidth / 2;
  gridHtml += '<div class="today-line" style="left:' + todayOffset + 'px"></div>';

  document.getElementById('labelBody').innerHTML = labelHtml;
  document.getElementById('gridBody').innerHTML = gridHtml;

  // 통계 업데이트
  updateStats(statCounts, todayOut, todayIn, filtered.length);

  // 매출 차트 렌더링
  renderRevenue(filtered, totalDays);

  // 스크롤 동기화 바인딩
  syncScroll();
}

function daysBetween(a, b) {
  return Math.round((b.getTime() - a.getTime()) / 86400000);
}

function emptyCells(count, today) {
  var html = '';
  var d = new Date(state.dateRange.start);
  for (var i = 0; i < count; i++) {
    var cls = 'grid-cell';

    if (d.getDay() === 0 || d.getDay() === 6) cls += ' weekend';
    html += '<div class="' + cls + '"></div>';
    d = new Date(d.getTime() + 86400000);
```

```javascript
function setupDrag() {
  var gridBody = document.getElementById('gridBody');
  var dragging = null;  // { el, itemId, startX, origLeft, origItem }

  gridBody.addEventListener('pointerdown', function(e) {
    var bar = e.target.closest('.timeline-bar');
    if (!bar || e.target.classList.contains('resize-handle')) return;
    e.preventDefault();
    bar.setPointerCapture(e.pointerId);

    var itemId = bar.getAttribute('data-item-id');
    var item = state.items.find(function(i) { return i.id === itemId; });
    if (!item) return;

    dragging = {
      el: bar,
      itemId: itemId,
      startX: e.clientX,
      origLeft: parseInt(bar.style.left),
      origItem: item
    };
    bar.style.cursor = 'grabbing';
    bar.style.opacity = '0.85';
    bar.style.zIndex = '20';
  });

  gridBody.addEventListener('pointermove', function(e) {
    if (!dragging) return;
    var dx = e.clientX - dragging.startX;
    dragging.el.style.left = (dragging.origLeft + dx) + 'px';
  });

  gridBody.addEventListener('pointerup', function(e) {
    if (!dragging) return;
    var dx = e.clientX - dragging.startX;
    var dayShift = Math.round(dx / state.colWidth);

    dragging.el.style.cursor = 'grab';
    dragging.el.style.opacity = '1';
    dragging.el.style.zIndex = '5';

    if (dayShift !== 0) {
      var item = dragging.origItem;
      var newStart = new Date(new Date(item.start).getTime() + dayShift * 86400000);
      var newEnd = new Date(new Date(item.end).getTime() + dayShift * 86400000);
      // 원래 시간 유지
      var origStart = new Date(item.start);
      var origEnd = new Date(item.end);
      newStart.setHours(origStart.getHours(), origStart.getMinutes());
      newEnd.setHours(origEnd.getHours(), origEnd.getMinutes());

      updateTimeAPI(item, newStart.toISOString(), newEnd.toISOString());
    } else {
      dragging.el.style.left = dragging.origLeft + 'px';
    }
    dragging = null;
  });
}
```

```javascript
      var newWidth = resizing.origWidth - dx;

        resizing.el.style.left = newLeft + 'px';
        resizing.el.style.width = newWidth + 'px';

      if (newWidth > 30) {
        resizing.el.style.width = newWidth + 'px';
      }
    }
  });

  gridBody.addEventListener('pointerup', function(e) {
    if (!resizing) return;
    var dx = e.clientX - resizing.startX;
    var dayShift = Math.round(dx / state.colWidth);

    if (dayShift !== 0) {
      var item = resizing.origItem;

      var newEnd = new Date(item.end);
```

`renderAll()` 안에서 바 렌더링 시, 같은 장비 + 겹치는 기간의 예약 수가 재고 초과인지 체크.

```javascript
function checkStockConflicts(items) {
  // 장비명 기준으로 기간 겹침 체크
  var conflicts = {};  // itemId → true
  var byEquip = {};    // 장비명 → [{ id, start, end, stock }]

  items.forEach(function(item) {
    var equip = item.장비명 || item.세트명 || '';
    if (!equip) return;
    if (!byEquip[equip]) byEquip[equip] = [];
    byEquip[equip].push({
      id: item.id,
      start: stripTime(new Date(item.start)).getTime(),
      end: stripTime(new Date(item.end)).getTime(),
      stock: item.재고 || 1
    });
  });

  Object.keys(byEquip).forEach(function(equip) {
    var list = byEquip[equip];
    if (list.length <= 1) return;
    var stock = list[0].stock;

    // 날짜별 예약 수 세기
    var dateCount = {};
    list.forEach(function(r) {
      for (var d = r.start; d <= r.end; d += 86400000) {
        dateCount[d] = (dateCount[d] || 0) + 1;
      }
    });

    // 재고 초과 날짜가 있으면 해당 날짜에 걸치는 모든 아이템 충돌
    Object.keys(dateCount).forEach(function(d) {
      if (dateCount[d] > stock) {
        list.forEach(function(r) {
          if (Number(d) >= r.start && Number(d) <= r.end) {
            conflicts[r.id] = true;
          }
        });
      }
    });
  });

  return conflicts;
}
```

```javascript
function renderRevenue(items, totalDays) {
  var chart = document.getElementById('revenueChart');
  if (!chart) return;

  // 일별 매출 합산
  var dailyRev = {};
  items.forEach(function(item) {
    if (!item.단가 || item.status === '취소') return;
    var s = stripTime(new Date(item.start));
    var e = stripTime(new Date(item.end));
    var days = daysBetween(s, e) + 1;
    var daily = item.단가 / Math.max(days, 1);  // 일할 계산
    for (var d = new Date(s); d <= e; d = new Date(d.getTime() + 86400000)) {
      var key = d.getTime();
      dailyRev[key] = (dailyRev[key] || 0) + daily;
    }
  });

  var maxRev = 0;
  Object.values(dailyRev).forEach(function(v) { if (v > maxRev) maxRev = v; });

  var html = '';
  var d = new Date(state.dateRange.start);
  for (var i = 0; i < totalDays; i++) {
    var rev = dailyRev[d.getTime()] || 0;
    var h = maxRev > 0 ? Math.round((rev / maxRev) * 100) : 0;
    var label = rev >= 10000 ? Math.round(rev / 10000) + '만' : (rev > 0 ? Math.round(rev / 1000) + '천' : '');
    html += '<div class="revenue-col" style="width:' + state.colWidth + 'px;min-width:' + state.colWidth + 'px">'
      + '<div class="revenue-bar" style="height:' + h + '%" title="' + Math.round(rev).toLocaleString() + '원"></div>'
```

```javascript
function setupAddBooking() {
  var gridBody = document.getElementById('gridBody');
  var addDragging = null;

  gridBody.addEventListener('pointerdown', function(e) {
    // 바나 리사이즈 핸들이 아닌 빈 영역에서만
    if (e.target.closest('.timeline-bar')) return;
    var cell = e.target.closest('.grid-cell');
    if (!cell) return;

    // 날짜 계산: cell의 형제 순서로 인덱스 파악
    var row = cell.parentElement;
    var cellIndex = Array.from(row.children).filter(function(c) {
      return c.classList.contains('grid-cell');
    }).indexOf(cell);

    addDragging = { startIndex: cellIndex, el: cell };
  });

  gridBody.addEventListener('pointerup', function(e) {
    if (!addDragging) return;
    var cell = e.target.closest('.grid-cell');
    var endIndex = addDragging.startIndex;
    if (cell) {
      var row = cell.parentElement;
      endIndex = Array.from(row.children).filter(function(c) {
        return c.classList.contains('grid-cell');
      }).indexOf(cell);
    }

    var fromIdx = Math.min(addDragging.startIndex, endIndex);
    var toIdx = Math.max(addDragging.startIndex, endIndex);
    var fromDate = new Date(state.dateRange.start.getTime() + fromIdx * 86400000);
    var toDate = new Date(state.dateRange.start.getTime() + toIdx * 86400000);

    addDragging = null;
    showAddModal(fromDate, toDate);
  });
}
```

```javascript
function setupPinchZoom() {
  var grid = document.getElementById('gridPanel');
  var lastDist = 0;

  grid.addEventListener('touchstart', function(e) {
    if (e.touches.length === 2) {
      lastDist = Math.hypot(
        e.touches[0].clientX - e.touches[1].clientX,
        e.touches[0].clientY - e.touches[1].clientY
      );
    }
  }, { passive: true });

  grid.addEventListener('touchmove', function(e) {
    if (e.touches.length !== 2) return;
    var dist = Math.hypot(
      e.touches[0].clientX - e.touches[1].clientX,
      e.touches[0].clientY - e.touches[1].clientY
    );
    if (lastDist > 0) {
      var delta = dist - lastDist;
      if (delta > 30) {
        // 줌 인
        document.getElementById('zoomIn').click();
        lastDist = dist;
      } else if (delta < -30) {
        // 줌 아웃
        document.getElementById('zoomOut').click();
        lastDist = dist;
      }
    }
    lastDist = dist;
  }, { passive: true });
}
```
